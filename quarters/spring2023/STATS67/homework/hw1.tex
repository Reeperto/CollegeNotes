\section*{Problem 5}
\subsection*{Part A}
\begin{align*}
	P(A^\complement B^\complement) &= P((A \cup B)^\complement) \\
																			&= 1 - P(A \cup B) \\
																			&= 1 - 0.8 \\
																			&= 20\%
\end{align*}
\subsection*{Part B}
\begin{align*}
	P(A\cup B) &= P(A) + P(B) - P(AB) \\
	P(AB) &= P(A) + P(B) - P(A \cup B) \\
	&= 0.5 + 0.45 - 0.8 \\
	&= 15\%
\end{align*}
\subsection*{Part C}
\begin{align*}
	P(AB^\complement) &= P(A) - P(AB) \\
										&= 0.5 - 0.15 \\
										&= 35\%
\end{align*}
